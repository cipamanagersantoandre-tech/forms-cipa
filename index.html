function openResponsesView(formId) { const formData = forms.find(f => f.id === formId); if (!formData) return; // FORÇAR detecção do parâmetro const urlParams = new URLSearchParams(window.location.search); const hasFormParam = urlParams.has('form'); const formParamValue = urlParams.get('form'); const isPublic = hasFormParam && formParamValue === formId; console.log('🔍 DEBUG:', { url: window.location.href, hasFormParam: hasFormParam, formParamValue: formParamValue, formId: formId, isPublic: isPublic }); currentFormId = formId; document.querySelectorAll('.view').forEach(v => v.classList.add('hidden')); document.getElementById('view-responses').classList.remove('hidden'); document.getElementById('responses-form-title').textContent = formData.name; if (isPublic) { console.log('✅ Modo PÚBLICO ativado'); renderPublicForm(); } else { console.log('✅ Modo ADMIN ativado'); renderResponsesDashboard(); } }
